Watch(() => // Signature: Watch(Func<string>) // just another example
	{
		StringBuilder sb = new StringBuilder();
		var agents = Player.PlayerManager.PlayerAgentsInLevel;
		for (int i = 0; i < agents.Count - 1; i++) {
			var agent = agents[i];
			sb.Append(agent.Owner.NickName);
			sb.Append(", ");
		}
		sb.Append(agents[agents.Count-1].Owner.NickName);
		return sb.ToString();
	}
);

foreach (var player in Player.PlayerManager.PlayerAgentsInLevel)

			return $"{obj.Owner.NickName} - {obj.Position} - {obj.Inventory.WieldedItem?.PublicName ?? "EmptyHands"}";
		}
	);
}
